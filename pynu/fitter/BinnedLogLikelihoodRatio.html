  6class BinnedLogLikelihoodRatio:
  7    r"""Class containing all the information needed to perform an analysis and the methods for computing
  8    the log likelihood ratio ($-2\ln\big(\frac{L(Exp.)}{L(Obs.)}\big)\sim\chi^2$) given a set of binned
  9    observed data, binned expected events at a given physics point and nuisance parameters, and assuming
 10    Poisson statistics.
 11   """
 12
 13    def __init__(self, observation, nominal_nuisance,
 14                 sigma_nuisance,
 15                 dist_nuisance):
 16        r"""Initiates the class by storing the non-changing items of the $\chi^2$ calculation.
 17
 18        Args:
 19            observation (dict): Produced by PyNuFit and follows the structue (Experiment(str): binned events (numpy.array).
 20            nominal_nuisance (list of float): Produced from the xml analysis file, it contains the nominal values assumed
 21            for the nuisance parameters.
 22            sigma_nuisance (list of float): Produced from the xml analysis file, it contains the standard deviation values
 23            assumed for the nuisance parameters.
 24            dist_nuisance (list of str): Produced from the xml analysis file, it contains the type of distribution which is
 25            assumed for each nuisance.
 26
 27       """
 28        self.observation = observation
 29        self.nominal_nuisance = nominal_nuisance
 30        self.sigma_nuisance = sigma_nuisance
 31        self.dist_nuisance = dist_nuisance
 32        self.number_of_nuisance = len(self.nominal_nuisance)
 33
 34    def stats_only(self, expectation):
 35        r"""Returns the value of binned $\chi^2 = 2\sum_i \Big(
 36        E_i-O_i+O_i\ln\big(\frac{O_i}{E_i}\big)\Big)$, given the dictionary of binned expected number of events
 37        for each experiment of the analysis.
 38
 39        Args:
 40            expectation (dict): Produced by `pynu.PyNuFit` and follows the structue (Experiment(str): binned events
 41            (numpy.array): similarly to observation, but for a given physics and nuisance values.
 42
 43        Returns:
 44            Float with the value of $\chi^2$.
 45       """
 46        X2: float = 0
 47        for O, E in zip(self.observation.values(),
 48                        expectation.values()):
 49            X2 += 2 * np.sum(E - O + O * np.log(O / E))
 50        return X2

 52    def stats_and_systematics(
 53            self,
 54            expectation,
 55            nuisance):
 56        r"""Returns the value of binned $\chi^2 = 2\sum_i \Big(
 57        E_i-O_i+O_i\ln\big(\frac{O_i}{E_i}\big)\Big) + 2\sum_j \ln\Big(\frac{P^{nuis}_j(x)}{P^{nuis}_j(x=\mu)}\Big)$,
 58        given the dictionary of binned expected number of events for each experiment of the analysis and taking
 59        into account the nuisance penalty terms.
 60
 61        Args:
 62            expectation (dict): Produced by PyNuFit and follows the structue (Experiment(str): binned events
 63            (numpy.array) similarly to observation, but for a given physics and nuisance values.
 64            nuisance (list of float): Values for the nuisance parameters ordered as provided by ParseXML class.
 65
 66        Returns:
 67            Float with the value of $\chi^2$ with nuisance.
 68       """
 69        X2: float
 70        if set(nuisance) == set(self.nominal_nuisance):
 71            X2 = self.stats_only(expectation)
 72        else:
 73            X2 = self.stats_only(expectation) + \
 74                self.nuisance_penalty(nuisance)
 75        return X2

 77    def gradient(
 78            self,
 79            expectation,
 80            diff_expectation,
 81            nuisance):
 82        r"""Returns the gradient of binned $\chi^2$ computed analytically, given the dictionary of binned
 83        expected number of events for each experiment of the analysis and its derivative with respect to every
 84        nuisance parameter.
 85
 86        $\nabla_j \chi^2 = 2~\sum_{i} \Big( 1 - \frac{O_i}{E_i}\Big)\frac{\partial E_i}{\partial x_j} + \frac{2}{P^{nuis}_j(x)} \frac{d~P^{nuis}_j(x)}{dx_j}$
 87
 88        LIMITATION: Currently, this is only done for nuisance following normal distributions. Other distributions
 89        like Beta will come soon.
 90
 91        Args:
 92            expectation (dict): Produced by `pynu.PyNuFit` and follows the structue (Experiment(str): binned events
 93            (numpy.array) similarly to observation, but for a given physics and nuisance values.
 94            diff_expectation (dict): Produced by `pynu.PyNuFit` and follows the structue (nuisance parameter (str):
 95            (Experiment(str): binned events (numpy.array)).
 96            nuisance (list of float): Values for the nuisance parameters ordered as provided by ParseXML class.
 97
 98        Returns:
 99            Numpy array with each component of $\nabla \chi^2$.
100       """
101        nabla_X2 = np.zeros(len(self.nominal_nuisance))
102        for i, (dE, mu, sig, dist, nuis) in enumerate(zip(diff_expectation.values(
103        ), self.nominal_nuisance, self.sigma_nuisance, self.dist_nuisance, nuisance)):
104            if dist == 'normal':
105                nabla_X2[i] += diff_log_gaussian_ratio(nuis, mu, sig)
106            elif dist == 'beta':
107                nabla_X2[i] += diff_log_beta_ratio(nuis, mu, sig)
108            else:
109                sys.exit(
110                    f'Not an implemented distribution for nuisance {list(diff_expectation.keys())[i]}.')
111
112            for O, E, dEdx in zip(self.observation.values(),
113                                  expectation.values(), dE.values()):
114                nabla_X2[i] += 2 * np.sum((1 - O / E) * dEdx)
115
116        print(nabla_X2)
117
118        return nabla_X2

142    def analytic_priors_bounds(self,
143                               expectation,
144                               diff_expectation):
145        r"""Returns the first-order values of the nuisance parameters which minimize the $\chi^2$ at a given
146        physics points. Here, first-order means we assume that the binned expected number of events is not
147        modified by nuisance parameters, i.e. nuisance parameters are assumed to take the default value in this
148        approximation.
149
150        $\nabla_j \chi^2  =0$, and at first order, $E'_i \approx E_i + \frac{\partial E_i}{\partial x_j} (x_j-\mu_j)$,
151        where $E_i$ is the number of expected events with nuisance at their nomnial values.
152
153        $\widetilde{x_j} = \mu_j + \frac{\sum \Big(\frac{O_i}{E_i} - 1 \Big) \left.\frac{\partial~E_i}{\partial x_j}\right\vert_{x_j=\mu_j} } {\sum \frac{O_i}{{E_i}^2} \Big( \left.\frac{\partial~E_i}{\partial x_j}\right\vert_{x_j=\mu_j}\Big)^2 + \frac{1}{\sigma^2_j}}$
154
155        Further, bounds for the final values of the nuisance parameters as follows.
156
157        $x_j\in[\widetilde{x_j}-\delta_j,\widetilde{x_j}+\delta_j]$, where $\delta_j = \min(2\cdot|\widetilde{x_j} - \mu_j|, \sigma)$
158
159        All this information is very useful for the minimizer to find faster the values of nuisance parameters
160        minimizing the $\chi^2$.
161
162        Args:
163            expectation (dict): Produced by `pynu.PyNuFit` and follows the structue (Experiment(str): binned events
164            (numpy.array) similarly to observation, but for a given physics and nuisance values.
165            diff_expectation (dict): Produced by `pynu.PyNuFit` and follows the structue (nuisance parameter (str):
166            (Experiment(str): binned events (numpy.array)).
167
168        Returns:
169            Numpy array with the estimate for the nuisance parameters.
170            Tuple with the lower and upper bounds for the nuisance parameters. Tuple(Tuple(lower,upper)).
171       """
172        A = np.zeros(self.number_of_nuisance)
173        B = np.zeros(self.number_of_nuisance)
174        mu = np.array(self.nominal_nuisance)
175        sig = np.array(self.sigma_nuisance)
176
177        # Experiments
178        for i, dE in enumerate(diff_expectation.values()):
179            for O, E, dEdx in zip(self.observation.values(),
180                                  expectation.values(), dE.values()):
181                A[i] += np.sum((O / E - 1) * dEdx)
182                B[i] += np.sum(O / E**2 * dEdx**2)
183
184        # Missing non-normal distribution cases
185        priors = mu + A / (B + 1 / sig**2)
186
187        delta = np.minimum(2 * np.abs(priors - mu), sig)
188        delta[delta == 0] = sig[delta == 0]
189
190        bounds = np.c_[priors - 2 * delta, priors + 2 * delta]
191        bounds = tuple(map(tuple, bounds))
192
193        self.diff_expectation_nominal = diff_expectation
194        self.expectation_nominal = expectation
195        print(priors)
196        print(bounds)
197
198        return priors, bounds

200    def parabolic_priors(self,
201                         expectation_prior,
202                         diff_expectation_prior,
203                         prior):
204        r"""Second order analytic computation of values for parameters to be mariginalized
205        assuming we are close enough to the minimum , i.e. a parabola, i.e. linear derivative.
206
207        $\widetilde{x_j} = \frac{\nabla_j \chi^2(x'_j) * \mu_j - \nabla_j \chi^2(\mu_j) * x'_j}{\nabla_j \chi^2(x'_j) - \nabla_j \chi^2(\mu_j)}$
208
209        NOTE: This method does not work well and needs more thought so use it carefully.
210
211        Args:
212            expectation (dict): Produced by `pynu.PyNuFit` and follows the structue (Experiment(str): binned events
213            (numpy.array) similarly to observation, but for a given physics and nuisance values.
214            diff_expectation (dict): Produced by `pynu.PyNuFit` and follows the structue (nuisance parameter (str):
215            (Experiment(str): binned events (numpy.array)).
216            priors (list of float): Values of nuisance parameter estimates other than the nominal values.
217
218        Returns:
219            Numpy array with the estimate for the nuisance parameters.
220
221        """
222
223        if not self.expectation_nominal:
224            sys.exit('Expectation for nominal values of nuisance not defined.')
225
226        if not self.diff_expectation_nominal:
227            sys.exit(
228                'Derivative of the expectation for nominal values of nuisance not defined.')
229
230        D_Chi2_1 = self.gradient(
231            expectation_prior,
232            diff_expectation_prior,
233            prior)
234        X_1 = prior
235
236        D_Chi2_0 = self.gradient(
237            self.expectation_nominal,
238            self.diff_expectation_nominal,
239            self.nominal_nuisance)
240        X_0 = np.array(self.nominal_nuisance)
241
242        priors_2nd = (D_Chi2_1 * X_0 - D_Chi2_0 * X_1) / (D_Chi2_1 - D_Chi2_0)
243
244        return priors_2nd
