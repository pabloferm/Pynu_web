122    def BinIt_MC_1D(self, array):  # 1D energy binning
123        for hist in self.Binner:
124            hist.reset()
125
126        if self.scale_E == 1 and self.bias_E == 0:
127            E = self.EReco
128        else:
129            E = self.EReco * self.scale_E + self.bias_E
130
131        v = np.array([])
132        for i, hist in enumerate(self.Binner):
133            v = np.hstack(
134                (v, hist.fill(
135                    E[self.Sample == i],
136                    weight=array[self.Sample == i] * self.BaseWeight
137                    [self.Sample == i]).values().reshape(-1)))
138
139        return v
