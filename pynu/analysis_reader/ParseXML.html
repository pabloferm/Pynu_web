 62        """Sets all analysis variables, that is all the sources, targets, detectors and oscillation
 63        parameters of the given analysis. It also computes the number of nuisance and physics parameters,

209    def physics_grid(self):
210        """Builds the arrays with all the physics points to be sampled from in the analysis.
211
212        Args:
213            None
214
215        Returns:
216            Dict with the structure {parameter (str): numpy.array}
217        """
218        phys_grid = {}
219        for par in self.Physics:
220            phys_grid[par] = {}
221            for j, item in enumerate(self.Physics[par]):
222                if item == 'Ordering':
223                    phys_grid[par][item] = np.array(self.PhysEdges[par][j])
224                elif item[-1].isdigit():
225                    if int(item[-1]) > 3:
226                        phys_grid[par][item] = np.geomspace(
227                            self.PhysEdges[par][j][0],
228                            self.PhysEdges[par][j][-1],
229                            self.PhysPoints[par][j])
230                    else:
231                        phys_grid[par][item] = np.linspace(
232                            self.PhysEdges[par][j][0],
233                            self.PhysEdges[par][j][-1],
234                            self.PhysPoints[par][j])
235                else:
236                    phys_grid[par][item] = np.linspace(
237                        self.PhysEdges[par][j][0],
238                        self.PhysEdges[par][j][-1],
239                        self.PhysPoints[par][j])
240        return phys_grid

242    def cartesian_physics_grid(self):
243        """Builds the grid of physics points to be sampled from in the analysis. It is computed as the cartesian
244        product of the 'self.PhysGrid' arrays and ordered in the same way.
245        the
246
247        Args:
248            None
249
250        Returns:
251            List of lists with all the pysics points.
252        """
253        v = []
254        for par in self.PhysGrid.keys():
255            for item, values in self.PhysGrid[par].items():
256                v.append(values)
257        return [*itertools.product(*v)]

330    def reader(self, item, atrib='name'):
331        """Main general method for reading a given item or block from the analysis xml file. It adds the
332        read information into the class variables classifying each of the tunes as fixed, nuisance or
333        physics as stated in the xml file.
334
335        Args:
336            item (str): It provides the block to be read. 'NeutrinoSource', 'NeutrinoTarget', etc.
337            atrib (str, optional): By default this variable is set to 'name' as it is the most common use
338            to read a whole block. However, further functionality is provided to read a given tune or
339            parameter in a block.
340
341        Returns:
342            None if atrib == 'name' and a list of items in the rest of the cases.
343        """
344        itemList = []
345        osc = item == 'NeutrinoOscillations'
346        for source in self.root.iter(item):
347            if atrib == 'name':
348                if int(source.find('status').text):
349                    sname = source.attrib['name']
350                    itemList.append(source.attrib[atrib])
351                    if item == 'NeutrinoOscillations':
352                        self.Flavors = int(source.find('flavors').text)
353                    elif item == 'NeutrinoExperiment':
354                        self.Experiments[sname] = {}
355                        self.ExpTarget[sname] = source.find(
356                            'target').attrib['name']
357                        for src in source.findall('source'):
358                            if int(src.find('status').text):
359                                self.Experiments[sname][src.attrib['name']] = {
360                                }
361                                MCFiles = []
362                                DataFiles = []
363                                for fi in src.findall('MCFiles'):
364                                    if int(fi.find('status').text):
365                                        MCFiles.append(fi.attrib['name'])
366                                for fi in src.findall('DataFiles'):
367                                    if int(fi.find('status').text):
368                                        DataFiles.append(fi.attrib['name'])
369                                Exposure = float(src.find('exposure').text)
370                                MCyears = float(src.find('MCexposure').text)
371                        self.Experiments[sname][src.attrib['name']] = {
372                            'MCFiles': MCFiles, 'TotalMCexposure': MCyears,
373                            'DataFiles': DataFiles, 'Exposure': Exposure}
374                    self.Nuisance[sname] = []
375                    self.NuisSigma[sname] = {}
376                    self.NuisNominal[sname] = {}
377                    self.NuisDistribution[sname] = {}
378                    for nuis in source.findall('nuisance'):
379                        if int(nuis.find('status').text):
380                            s = nuis.attrib['name']
381                            if s == 'Ordering':
382                                sys.exit(
383                                    'Neutrino mass ordering cannot be a nuisance parameter. Please, test both ordering hypotheses.')
384                            else:
385                                self.NuisSigma[sname][s] = float(
386                                    nuis.find('sigma').text)
387                                self.NuisSigmaList.append(
388                                    float(nuis.find('sigma').text))
389                                self.NuisNominal[sname][s] = float(
390                                    nuis.find('nominal').text)
391                                self.NuisNominalList.append(
392                                    float(nuis.find('nominal').text))
393                                self.NuisDistribution[sname][s] = str(
394                                    nuis.find('distribution').text).strip()
395                                self.NuisDistributionList.append(
396                                    str(nuis.find('distribution').text).strip())
397                                self.Nuisance[sname].append(s)
398                                self.NuisanceList.append(s)
399                    self.Fixed[sname] = []
400                    self.FixedValue[sname] = {}
401                    for fix in source.findall('fixed'):
402                        if int(fix.find('status').text):
403                            s = fix.attrib['name']
404                            if s == 'Ordering':
405                                self.FixedValue[sname][s] = fix.find(
406                                    'value').text
407                                self.FixedValueList.append(
408                                    fix.find('value').text)
409                                self.Fixed[sname].append(s)
410                                self.FixedList = np.append(self.FixedList, s)
411                            else:
412                                self.FixedValue[sname][s] = float(
413                                    fix.find('value').text)
414                                self.FixedValueList.append(
415                                    float(fix.find('value').text))
416                                self.Fixed[sname].append(s)
417                                self.FixedList.append(s)
418                    self.Physics[sname] = []
419                    self.PhysTrue[sname] = {}
420                    self.PhysPoints[sname] = []
421                    self.PhysEdges[sname] = []
422                    for phys in source.findall('physics'):
423                        s = phys.attrib['name']
424                        if s == 'Ordering':
425                            points = int(phys.find('points').text)
426                            no = 0
427                            io = 0
428                            if 'norm' in phys.find(
429                                    'min').text or 'norm' in phys.find('max').text:
430                                no = 1
431                            if 'inv' in phys.find(
432                                    'min').text or 'inv' in phys.find('max').text:
433                                io = 1
434                            if io + no == 2 and points == 2:
435                                self.PhysTrue[sname][s] = phys.find(
436                                    'true').text
437                                self.PhysTrueList.append(
438                                    phys.find('true').text)
439                                self.PhysPoints[sname].append(2)
440                                self.PhysPointsList.append(2)
441                                self.PhysEdges[sname].append(
442                                    ['normal', 'inverted'])
443                                self.Physics[sname].append(s)
444                                self.PhysicsList.append(s)
445                            elif io + no == 1 and points == 1:
446                                if io:
447                                    self.FixedValue[sname][s] = 'inverted'
448                                    self.FixedValueList.append('inverted')
449                                    self.Fixed[sname].append(s)
450                                    self.FixedList.append(s)
451                                elif no:
452                                    self.FixedValue[sname][s] = 'normal'
453                                    self.FixedValueList.append('normal')
454                                    self.Fixed[sname].append(s)
455                                    self.FixedList.append(s)
456                                print(
457                                    'Notice: Parameter ' +
458                                    str(s) +
459                                    ' has been moved to fixed.')
460                            elif io + no == 0 or points == 0:
461                                sys.exit(
462                                    'Please, specify a neutrino mass ordering')
463                            else:
464                                sys.exit(
465                                    'Please, take a look to the Ordering, something is not well defined.')
466                        else:
467                            if float(
468                                    phys.find('min').text) == float(
469                                    phys.find('max').text) or int(
470                                    phys.find('points').text) <= 1:  # this parameter should be fixed
471                                self.FixedValue[sname][s] = float(
472                                    phys.find('true').text)
473                                self.FixedValueList.append(
474                                    float(phys.find('true').text))
475                                self.Fixed[sname].append(s)
476                                self.FixedList.append(s)
477                                print(
478                                    'Notice: Parameter ' +
479                                    str(s) +
480                                    ' has been moved to fixed.')
481                            elif float(phys.find('min').text) == float(phys.find('max').text):
482                                sys.exit('Please, check parameter ' + str(s))
483                            else:
484                                self.PhysTrue[sname][s] = float(
485                                    phys.find('true').text)
486                                self.PhysTrueList.append(
487                                    float(phys.find('true').text))
488                                self.PhysPoints[sname].append(
489                                    int(phys.find('points').text))
490                                self.PhysPointsList.append(
491                                    int(phys.find('points').text))
492                                self.PhysEdges[sname].append([
493                                    float(phys.find('min').text),
494                                    float(phys.find('max').text)])
495                                self.Physics[sname].append(s)
496                                self.PhysicsList.append(s)
497            else:
498                itemList.append(source.attrib[atrib])
499        return itemList

422                    self.PhysEdges[sname] = []

426                            points = int(phys.find('points').text)

430                                    'min').text or 'norm' in phys.find('max').text:

432                            if 'inv' in phys.find(
433                                    'min').text or 'inv' in phys.find('max').text:
434                                io = 1

436                                self.PhysTrue[sname][s] = phys.find(
437                                    'true').text

443                                    ['normal', 'inverted'])

498            else:

Main general method for reading a given item or block from the analysis xml file. It adds the read information into the class variables classifying each of the tunes as fixed, nuisance or physics as stated in the xml file.

Arguments:
  • item (str): It provides the block to be read. 'NeutrinoSource', 'NeutrinoTarget', etc.
  • atrib (str, optional): By default this variable is set to 'name' as it is the most common use
  • to read a whole block. However, further functionality is provided to read a given tune or
  • parameter in a block.
Returns:

None if atrib == 'name' and a list of items in the rest of the cases.
