179    def ApplyWeights(self, tag, vector=None):
180        if tag == 'Fixed':
181            labels = self.Analysis.Fixed
182            vec = self.Analysis.FixedValue
183        elif tag == 'Nominal':
184            labels = self.Analysis.Nuisance
185            vec = self.Analysis.NuisNominal
186        elif tag == 'True':
187            labels = self.Analysis.Physics
188            vec = self.Analysis.PhysTrue
189        elif tag == 'Physics':
190            labels = self.Analysis.Physics
191            v_id = self.Analysis.PhysicsList
192        elif tag == 'Nuisance':
193            labels = self.Analysis.Nuisance
194            v_id = self.Analysis.NuisanceList
195        else:
196            sys.exit('Not a valid tag for applying weights.')
197
198        for name, exp in self.Experiments.items():
199            for source in labels:
200                if source in exp.Definition.keys():
201                    tune_block = exp.Definition[source]
202                    for tune in labels[source]:
203                        if vector is not None:
204                            idx = v_id.index(tune)
205                            value = vector[idx]
206                        else:
207                            value = vec[source][tune]
208
209                        if tune_block == 'Flux':
210                            w = self.PhysicsTunes[name].GetFlux(tune, value)
211                        elif tune_block == 'XSection':
212                            w = self.PhysicsTunes[name].GetXSection(
213                                tune, value)
214                        elif tune_block == 'Detector':
215                            w = self.PhysicsTunes[name].GetDetector(
216                                tune, value)
217                        elif tune_block == 'Osc':
218                            self.PhysicsTunes[name].OscillationTunes.UpdateParameter(
219                                tune, value)
220
221                        if tune_block != 'Osc':
222                            if tag == 'Fixed':
223                                exp.UpdateBaseWeights(w)
224                            elif tag in ['True', 'Nominal']:
225                                exp.UpdateNominalWeights(w)
226                            elif tag == 'Physics':
227                                exp.UpdatePhysicsWeights(w)
228                            elif tag == 'Nuisance':
229                                exp.UpdateNuisanceWeights(w)

231    def GetDiffLogWeights(self, vector):
232        ''' Computes the derivative with respect the nuisance parameter nuis '''
233        ''' Returns a dict of nuis : experiment : partial of weight with respect to nuis over weight '''
234        dWoverW = {}
235
236        for source, nuisance_list in self.Analysis.Nuisance.items():
237            for name, exp in self.Experiments.items():
238                if source in exp.Definition.keys():
239                    tune_block = exp.Definition[source]
240                    for tune in self.Analysis.Nuisance[source]:
241                        dWoverW[tune] = {name: 0}
242                        idx = self.Analysis.NuisanceList.index(tune)
243                        if tune_block == 'Flux':
244                            dWoverW[tune][name] = self.PhysicsTunes[name].GetFlux(
245                                'diff_' + tune, vector[idx]) / self.PhysicsTunes[name].GetFlux(tune, vector[idx])
246                        elif tune_block == 'XSection':
247                            dWoverW[tune][name] = self.PhysicsTunes[name].GetXSection(
248                                'diff_' + tune, vector[idx]) / self.PhysicsTunes[name].GetXSection(tune, vector[idx])
249                        elif tune_block == 'Detector':
250                            dWoverW[tune][name] = self.PhysicsTunes[name].GetDetector(
251                                'diff_' + tune, vector[idx]) / self.PhysicsTunes[name].GetDetector(tune, vector[idx])
252                        elif tune_block == 'Osc':
253                            dWoverW[tune][name] = self.PhysicsTunes[name].GetOscillation(
254                                'diff_' + tune, vector[idx]) / self.PhysicsTunes[name].OscillationTunes.GetOscillations()
255        return dWoverW

267    def FitModel(self, point, mode='BinnedLogLikelihoodRatio', method='L-BFGS-B'):  # 'SLSQP' 'GD' 'ADAM'
268        ''' Binned log-Likelihood fit assuming data is Poisson-distributed '''
269        self.set_likelihood(mode)
270        self.point = point
271
272        ''' Binned log-Likelihood fit assuming data is Poisson-distributed '''
273        self.ComputeBinnedExpectation(
274            self.point, physics=True)  # Nominal expectation
275        ''' Statistics only computation to start guiding the minimization '''
276        X2_stats = self.LLH.stats_only(self.Expectation)
277
278        '''Get Jacobian of expected events w.r.t. nuisance parameters'''
279        self.ComputeBinnedDiffExpectation()
280
281        '''Analytic estimate for priors and bounds at first order'''
282        AnalyticPrior, AnalyticBounds = self.LLH.analytic_priors_bounds(
283            self.Expectation, self.DiffExpectation)
284
285        print(type(AnalyticPrior))
286        '''Combined chi^2 minimization'''
287        if method == 'GD':
288            from .gradient_descent_minimizer import gradient_descent_minimizer
289            gradient_descent_minimizer(
290                self.model_tester,
291                AnalyticPrior,
292                bounds=AnalyticBounds)
293        elif method == 'ADAM':
294            from .adam_minimizer import adam_minimizer
295            adam_minimizer(
296                self.model_tester,
297                AnalyticPrior,
298                bounds=AnalyticBounds)
299        elif method == 'migrad':
300            import iminuit
301            res = iminuit.minimize(
302                self.model_tester,
303                AnalyticPrior,
304                method=method,
305                jac=True,
306                bounds=AnalyticBounds,
307                options={
308                    'disp': self.verbosity})
309        else:
310            res = minimize(
311                self.model_tester,
312                AnalyticPrior,
313                method=method,
314                jac=True,
315                bounds=AnalyticBounds,
316                options={
317                    'disp': self.verbosity})
318
319        self.WriteToOutFile(
320            self.point,
321            'Analysis',
322            'Chi2 Stats. Only',
323            X2_stats)
324
325        self.WriteToOutFile(
326            self.point,
327            'Nuisance Parameters',
328            self.Analysis.NuisanceList,
329            res.x.tolist())
330
331        self.WriteToOutFile(self.point, 'Analysis', 'Chi2 Systs.', res.fun)
332
333        if self.verbosity:
334            print(f'Fitted nuisances: {res.x}')
335            print(f'-2 ln(H/H0) = {res.fun}')
336
337        return - 0.5 * res.fun

339    def model_tester(self, nuisance_vector):
340        ''' Compute expected and its derivatives '''
341        self.ComputeBinnedExpectation(
342            self.point, nuisance_vector=nuisance_vector)  # Nominal expectation
343        self.ComputeBinnedDiffExpectation(nuisance_vector=nuisance_vector)
344
345        ''' Get -2 ln(H/H0) ~ χ2 '''
346        Chi2 = self.LLH.stats_and_systematics(
347            self.Expectation,
348            nuisance_vector)
349
350        ''' The gradient of the above '''
351        D_Chi2 = self.LLH.gradient(
352            self.Expectation,
353            self.DiffExpectation,
354            nuisance_vector)
355        print(f'chi-squared = {Chi2}')
356        print(f'gradient of chi-squared = {D_Chi2}')
357        print(f'norm of gradient of chi-squared = {np.linalg.norm(D_Chi2)}')
358
359        return (Chi2, D_Chi2)

389                    idx = self.Analysis.PhysicsList.index(par)
390                    this.create_dataset(
391                        par, data=physics_lists[idx],
392                        compression='gzip')
393
394            grp = hf.create_group('Analysis')
395            grp.create_dataset(
396                'Chi2 Stats. Only',
397                data=[0.0] *
398                self.Analysis.NumberOfPhysPoints,
399                compression='gzip')
400            if self.Analysis.wSyst:
401                grp.create_dataset(
402                    'Chi2 Systs.',
403                    data=[0.0] *
404                    self.Analysis.NumberOfPhysPoints,
405                    compression='gzip')
